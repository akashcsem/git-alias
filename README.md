# git-alias
``

alias prev='cd -'
alias home='cd ~/'

alias art='php artisan $*'
alias serve='php artisan serve'
alias tinker='php artisan tinker'
alias fresh='php artisan migrate:fresh'
alias seed='php artisan db:seed'
alias request='php artisan make:request $*'
alias model='php artisan make:model $*'
alias controller='php artisan make:controller $*'
alias migration='php artisan make:migration $*'
alias seeder='php artisan make:seeder $*'

alias checkout='git checkout $1'
alias merge='git merge $1'
alias push='git push'
alias add='git add $*'
alias status='git status'
alias fetch='git fetch'
alias pull='git pull'
alias log='git log'
alias commit='git commit -m $1'
alias commitall='git add -A && git commit -m $1'
alias uncommit='git reset HEAD~1 --soft'
alias unstage='git reset HEAD $*' 
alias commitfiles='git log -1 --stat'

alias autoload='composer dump-autoload'
alias servefresh='php artisan migrate:fresh && php artisan db:seed && php artisan serve'
alias serveauto='composer dump-autoload && php artisan migrate:fresh && php artisan db:seed && php artisan serve'

alias q='exit'
alias back='cd ..'
alias ..='cd ..'
alias akash='cd akash'
alias skycoder='git checkout skycoder'
alias master='git checkout master'
alias makash='git merge akash'
alias mskycoder='git merge skycoder'
alisa mmaster='git merge skycoder'
alias mmaster='git merge master'
``
